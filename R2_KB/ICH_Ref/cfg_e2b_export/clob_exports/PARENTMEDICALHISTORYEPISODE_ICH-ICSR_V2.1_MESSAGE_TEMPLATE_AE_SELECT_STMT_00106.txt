SELECT DECODE(CONDITION,NULL,NULL,Esm_Utl.GET_MEDDRA_VERSION(ITEM_DICT,2)) PARENTMDEPISODEMEDDRAVERSION,
  DECODE(Esm_Utl.GET_MEDDRA_VERSION(ITEM_DICT,2),NULL,NULL,CONDITION) PARENTMEDICALEPISODENAME,
  DECODE(START_DATE,NULL,NULL,DECODE(START_DATE_RES,5,602,7,610,102)) PARENTMEDICALSTARTDATEFORMAT,  DECODE(START_DATE_RES,5,TO_CHAR(START_DATE,'YYYY'),7,TO_CHAR(START_DATE,'YYYYMM'),8,TO_CHAR(START_DATE,'YYYYMMDD'),null) PARENTMEDICALSTARTDATE,
  CASE WHEN CONTINUE = 0 THEN 2 WHEN CONTINUE = 2 THEN 3 ELSE CONTINUE END PARENTMEDICALCONTINUE,
  DECODE(STOP_DATE,NULL,NULL,DECODE(STOP_DATE_RES,5,602,7,610,102)) PARENTMEDICALENDDATEFORMAT,  DECODE(STOP_DATE_RES,5,TO_CHAR(STOP_DATE,'YYYY'),7,TO_CHAR(STOP_DATE,'YYYYMM'),8,TO_CHAR(STOP_DATE,'YYYYMMDD'),NULL) PARENTMEDICALENDDATE,
  SUBSTR(NOTE,1,100) PARENTMEDICALCOMMENT
FROM CASE_PAT_HIST A,
  LM_CONDITION_TYPE B
WHERE A.CASE_ID                                                          = :CASE_ID
AND A.PARENT                                                             = 1
AND A.CONDITION_TYPE_ID                                                  = B.CONDITION_TYPE_ID(+)
AND NVL(B.COND_CATEGORY,1)                                               = 1
AND DECODE(Esm_Utl.GET_MEDDRA_VERSION(ITEM_DICT,2),NULL,NULL,CONDITION) IS NOT NULL
ORDER BY SORT_ID
