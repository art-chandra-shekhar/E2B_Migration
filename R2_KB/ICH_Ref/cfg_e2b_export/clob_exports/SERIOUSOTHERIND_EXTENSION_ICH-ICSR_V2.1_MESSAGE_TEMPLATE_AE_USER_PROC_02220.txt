DECLARE
  v_xml				varchar2(32767);
  l_return			number := 0;
  l_value			number;
  v_error_message	varchar2(4000);
  v_stored_procedure varchar2(30) := 'CREATE_CASE_EVENT';
BEGIN
  v_xml    := trim(ESM_IMP.F_READ_EXTENSION(:REPORT_ID,:DTD_ELEMENT));
  if v_xml is not null then
    if v_xml in ('0','1') then 
	   l_value := v_xml;
       l_return :=  ESM_IMP.F_WRITE(:REPORT_ID,:PARENT_ELEMENT,:DTD_ELEMENT,:PROFILE,'CASE_EVENT','SC_OTHER',l_value);
    else
	   v_error_message := 'Acceptable value for the dtd element SERIOUSOTHERIND_EXTENSION is 0,1. Module : '||v_stored_procedure;
	   ESM_IMP_UTL.WRITE_ERROR_MSG('SERIOUSOTHERIND_EXTENSION','','',v_error_message,:REPORT_ID);
    end if;
  end if;
END;
