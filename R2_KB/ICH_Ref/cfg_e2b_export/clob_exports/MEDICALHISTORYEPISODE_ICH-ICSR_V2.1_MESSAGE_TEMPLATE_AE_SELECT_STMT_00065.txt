SELECT DECODE(CONDITION,NULL,NULL,Esm_Utl.GET_MEDDRA_VERSION(ITEM_DICT,2)) PATIENTEPISODENAMEMEDDRAVERSIO,
  DECODE(Esm_Utl.GET_MEDDRA_VERSION(ITEM_DICT,2),NULL,NULL,CONDITION) PATIENTEPISODENAME,
  DECODE(START_DATE,NULL,NULL,DECODE(START_DATE_RES,5,602,7,610,102)) PATIENTMEDICALSTARTDATEFORMAT,  DECODE(START_DATE_RES,5,TO_CHAR(START_DATE,'YYYY'),7,TO_CHAR(START_DATE,'YYYYMM'),8,TO_CHAR(START_DATE,'YYYYMMDD'),NULL) PATIENTMEDICALSTARTDATE,
  CASE WHEN CONTINUE = 0 THEN 2 WHEN CONTINUE = 2 THEN 3 ELSE CONTINUE END PATIENTMEDICALCONTINUE,
  DECODE(STOP_DATE,NULL,NULL,DECODE(STOP_DATE_RES,5,602,7,610,102)) PATIENTMEDICALENDDATEFORMAT,  DECODE(STOP_DATE_RES,5,TO_CHAR(STOP_DATE,'YYYY'),7,TO_CHAR(STOP_DATE,'YYYYMM'),8,TO_CHAR(STOP_DATE,'YYYYMMDD'),NULL) PATIENTMEDICALENDDATE,
  SUBSTR(NOTE,1,100) PATIENTMEDICALCOMMENT
FROM CASE_PAT_HIST A,
  LM_CONDITION_TYPE B
WHERE A.CASE_ID                                                          = :CASE_ID
AND NVL(A.PARENT,0)                                                      = 0
AND A.CONDITION_TYPE_ID                                                  = B.CONDITION_TYPE_ID(+)
AND NVL(B.COND_CATEGORY,1)                                               = 1
and DECODE(ESM_UTL.GET_MEDDRA_VERSION(ITEM_DICT,2),null,null,CONDITION) is not null
ORDER BY SORT_ID
