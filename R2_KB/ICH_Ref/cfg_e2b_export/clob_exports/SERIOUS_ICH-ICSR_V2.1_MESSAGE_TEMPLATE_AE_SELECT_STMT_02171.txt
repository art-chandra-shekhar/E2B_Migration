SELECT
  CASE
    WHEN COUNT(1) > 0
    THEN 1
    ELSE 2
  END SERIOUS
FROM CASE_EVENT CE,
  CASE_ASSESS CA,
  GTT_ESM_REPORTABLE_EVENTS RE
WHERE CE.CASE_ID   = :CASE_ID
AND RE.REPORTABLE  = 1
AND CE.CASE_ID     = RE.CASE_ID
AND CE.SEQ_NUM     = RE.EVENT_SEQ_NUM
AND CA.CASE_ID     = CE.CASE_ID
AND (CE.SC_OTHER   = 1
OR CE.med_serious  = 1
OR CE.sc_int_req   = 1
OR CE.SC_DEATH     = 1
OR CE.SC_CONG_ANOM = 1
OR CE.SC_THREAT    = 1
OR CE.SC_DISABLE   = 1
OR CE.SC_HOSP      = 1)
AND CA.SERIOUSNESS = 1
