DECLARE
  v_xml					varchar2(32767);
  l_return				number := 0;
  v_error_message		varchar2(4000);
  v_stored_procedure	varchar2(30) := 'CREATE_CASE_PRODUCT';
  l_value				number;
  v_prod_name			varchar2(32767);
  lb_return				boolean := FALSE;
  v_dtd_elements		esm_imp.vtab;  
BEGIN
  v_xml    := trim(ESM_IMP.F_READ_EXTENSION(:REPORT_ID,:DTD_ELEMENT));
  if v_xml is not null then
   if v_xml in ('0','1') then 
		v_dtd_elements(1) := 'DRUGCHARACTERIZATION';
		v_dtd_elements(2) := 'MEDICINALPRODUCT';
		v_dtd_elements(3) := 'DRUGDOSAGEFORM';
		v_prod_name  := ESM_IMP.F_READ(:REPORT_ID,'MEDICINALPRODUCT');
		if v_prod_name is null then
			v_dtd_elements(2) := 'ACTIVESUBSTANCENAME';
		end if;
		lb_return := ESM_IMP.F_CHECK_AND_BUILD_KEY(:REPORT_ID,:PARENT_ELEMENT,v_dtd_elements);    
	   l_value := v_xml;
       l_return :=  ESM_IMP.F_WRITE(:REPORT_ID,:PARENT_ELEMENT,:DTD_ELEMENT,:PROFILE,'CASE_PROD_DEVICES','REPROCESSED_REUSE',l_value);
    else
	   v_error_message := 'Acceptable value for the dtd element REPROCESSEDSINGLEUSE_EXTENSION is 0,1. Module : '||v_stored_procedure;
	   ESM_IMP_UTL.WRITE_ERROR_MSG('REPROCESSEDSINGLEUSE_EXTENSION','','',v_error_message,:REPORT_ID);
    end if;
  end if;
END;
