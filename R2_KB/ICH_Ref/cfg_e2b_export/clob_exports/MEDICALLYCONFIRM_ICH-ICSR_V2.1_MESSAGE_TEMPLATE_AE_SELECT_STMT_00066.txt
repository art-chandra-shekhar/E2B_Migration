SELECT
  CASE
    WHEN CM.MEDICALLY_CONFIRM = 1
    THEN 1
    WHEN CM.MEDICALLY_CONFIRM = 0
    THEN 2
    WHEN RCE.EVENT_MEDICALLYCONFIRM IN(1, 2) AND :DIFFERENCE_RPT = 0
    THEN RCE.EVENT_MEDICALLYCONFIRM
    ELSE NULL
  END MEDICALLYCONFIRM
FROM CASE_MASTER CM,
  (SELECT
    CASE
      WHEN CE.MEDICAL_CONFIRM = 1
      THEN 1
      WHEN CE.MEDICAL_CONFIRM = 0
      THEN 2
      ELSE NULL
    END EVENT_MEDICALLYCONFIRM,
    CE.CASE_ID
  FROM CASE_EVENT CE,
    CASE_EVENT_ASSESS CEA,
    LM_CAUSALITY LC
  WHERE CE.CASE_ID        = :CASE_ID
  AND CEA.CASE_ID         = CE.CASE_ID
  AND CEA.EVENT_SEQ_NUM   = CE.SEQ_NUM
  AND CEA.DATASHEET_ID    = 0
  AND CEA.LICENSE_ID      = 0
  AND (( LC.CAUSALITY_ID  = CEA.DET_CAUSALITY_ID
  AND LC.REPORTABILITY    = 1)
  OR ( LC.CAUSALITY_ID    = CEA.RPT_CAUSALITY_ID
  AND LC.REPORTABILITY    = 1))
  AND CE.MEDICAL_CONFIRM IS NOT NULL
  AND CEA.DELETED        IS NULL
  ORDER BY EVENT_MEDICALLYCONFIRM
  ) RCE
WHERE CM.CASE_ID = :CASE_ID
AND CM.CASE_ID   = RCE.CASE_ID(+)
AND ROWNUM       = 1
