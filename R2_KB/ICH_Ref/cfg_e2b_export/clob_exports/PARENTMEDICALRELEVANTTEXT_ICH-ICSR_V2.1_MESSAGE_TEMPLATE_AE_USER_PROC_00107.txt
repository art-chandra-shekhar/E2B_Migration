DECLARE
   v_xml varchar2(32767);
   l_return number := 0;
BEGIN
   v_xml  := ESM_IMP.F_READ(:REPORT_ID,:DTD_ELEMENT);
   v_xml  := SUBSTR(v_xml,1,10000);
   if v_xml is not null then
      l_return :=  ESM_IMP.F_WRITE(:REPORT_ID,:PARENT_ELEMENT,:DTD_ELEMENT,:PROFILE,'CASE_PARENT_INFO','MED_HIST_TEXT',v_xml);
   end if;
END;
