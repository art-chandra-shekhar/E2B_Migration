SELECT PATIENTDEATHDATEFORMAT, 
  PATIENTDEATHDATE, 
  PATIENTAUTOPSYYESNO 
FROM 
  (SELECT 
DECODE(d.DEATH_DATE,NULL,NULL,DECODE(d.DEATH_DATE_RES,5,602,7,610,102)) 
PATIENTDEATHDATEFORMAT,      
DECODE(d.DEATH_DATE_RES,5,TO_CHAR(d.DEATH_DATE,'YYYY'),7,TO_CHAR(d.DEATH_DATE, 
'YYYYMM'),8,TO_CHAR(d.DEATH_DATE,'YYYYMMDD'),NULL) PATIENTDEATHDATE, 
    DECODE(d.AUTOPSY,10,1,9,1,8,1,4,2,12,3,NULL) PATIENTAUTOPSYYESNO 
  FROM CASE_DEATH D 
  WHERE d.CASE_ID = :CASE_ID 
  ) 
WHERE ((PATIENTDEATHDATEFORMAT IS NOT NULL 
AND PATIENTDEATHDATE           IS NOT NULL) 
OR PATIENTAUTOPSYYESNO         IS NOT NULL 
OR EXISTS 
  (SELECT 1 
  FROM CASE_DEATH_DETAILS 
  WHERE CASE_ID  = :CASE_ID 
  AND TERM_TYPE IN (1,2) 
  AND CAUSE     IS NOT NULL 
  ))
