SELECT DECODE(TEST_DATE,NULL,NULL,DECODE(TEST_DATE_RES,5,602,7,610,102)) TESTDATEFORMAT,
  DECODE(TEST_DATE_RES,5,TO_CHAR(TEST_DATE,'YYYY'),7,TO_CHAR(TEST_DATE,'YYYYMM'),8,TO_CHAR(TEST_DATE,'YYYYMMDD'),NULL) TESTDATE,
  SUBSTR(NVL(NVL(TEST_REPTD_J,TEST_REPTD),NVL(LAB_TEST_NAME_J,LAB_TEST_NAME)),1,100) TESTNAME,
  SUBSTR(NVL(RESULTS_J,RESULTS),1,50) TESTRESULT,
  DECODE(NVL(RESULTS_J,RESULTS),NULL,NULL,SUBSTR(NVL(L.UNIT_J,L.UNIT),1, 35)) TESTUNIT,
  SUBSTR(NORM_LOW,1,50) LOWTESTRANGE,
  SUBSTR(NORM_HIGH,1,50) HIGHTESTRANGE,
  ESM_UTL.GET_LAB_DATA_NOTES(:CASE_ID,SEQ_NUM,1) MOREINFORMATION
FROM CASE_LAB_DATA,
  (select unit_id,unit_j,unit from lm_dose_units where lab_test_unit=1) L
WHERE CASE_ID             = :CASE_ID
AND TEST_DATE            IS NOT NULL
AND CASE_LAB_DATA.UNIT_ID = L.UNIT_ID(+)
AND :RPT_CATEGORY NOT    IN ('E','F','G','L','M','N','O','P')
AND seq_num NOT          IN
  (SELECT SEQ_NUM
  FROM CASE_LAB_DATA A,
    (SELECT TEST_DATE,
      SUBSTR(NVL(NVL(TEST_REPTD_J,TEST_REPTD),NVL(LAB_TEST_NAME_J,LAB_TEST_NAME)),1,100) LAB_TEST_NAME_J,
      COUNT(1)
    FROM CASE_LAB_DATA
    WHERE CASE_ID     = :CASE_ID
    AND TEST_DATE_RES = 8
    HAVING COUNT(1)   > 1
    GROUP BY TEST_DATE,
      SUBSTR(NVL(NVL(TEST_REPTD_J,TEST_REPTD),NVL(LAB_TEST_NAME_J,LAB_TEST_NAME)),1,100)
    ) B
  WHERE A.TEST_DATE                                                                              = B.TEST_DATE
  AND SUBSTR(NVL(NVL(a.TEST_REPTD_J,a.TEST_REPTD),NVL(a.LAB_TEST_NAME_J,a.LAB_TEST_NAME)),1,100) = B.LAB_TEST_NAME_J
  AND A.TEST_DATE_RES                                                                            = 8
  AND A.CASE_ID                                                                                  = :CASE_ID
  )
ORDER BY SORT_ID
