DECLARE
  v_xml         varchar2(32767);
  l_return      number := 0;
  v_notes       varchar2(500);
  v_notes_j     varchar2(500);
BEGIN
      v_xml    := ESM_IMP.F_READ(:REPORT_ID,:DTD_ELEMENT);
      if v_xml = '1' then
          Begin
              SELECT TEXT, TEXT_J INTO v_notes, v_notes_j FROM LM_TRANSLATION_LABEL WHERE TRANSLATION_ID = 'ESM_MOREINFO';
              l_return :=  ESM_IMP.F_WRITE(:REPORT_ID,:PARENT_ELEMENT,:DTD_ELEMENT,:PROFILE,'CASE_LAB_DATA','NOTES',v_notes);
              IF :ARGUS_J = 1 THEN
                  l_return :=  ESM_IMP.F_WRITE(:REPORT_ID,:PARENT_ELEMENT,:DTD_ELEMENT,:PROFILE,'CASE_LAB_DATA','NOTES_J',v_notes_j);
              END IF;
          Exception
              When no_data_found Then NULL;
          End;
      end if;
END;
