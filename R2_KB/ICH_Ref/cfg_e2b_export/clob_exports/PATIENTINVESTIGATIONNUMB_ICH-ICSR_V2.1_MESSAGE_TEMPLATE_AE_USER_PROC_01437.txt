DECLARE
   v_xml varchar2(32767);
   l_return number := 0;
   v_privacy         varchar2(10) := 'PRIVACY';
BEGIN
   v_xml  := ESM_IMP.F_READ(:REPORT_ID,'PATIENTINVESTIGATIONNUMB');
   IF upper(v_xml) = v_privacy then
      v_xml := null;
   END IF;
   l_return :=  ESM_IMP.F_WRITE(:REPORT_ID,:PARENT_ELEMENT,:DTD_ELEMENT,:PROFILE,'CASE_PAT_INFO','PAT_SUBJ_NUM',v_xml);
END;
