DECLARE
  v_xml varchar2(32767);
  v_prod_name varchar2(32767);
  l_return number := 0;
  l_number      number;
  l_number2     number;
  lb_return     boolean := FALSE;
  v_dtd_elements esm_imp.vtab;
BEGIN
  v_xml  := ESM_IMP.F_READ(:REPORT_ID,'DRUGCHARACTERIZATION');
  if v_xml is not null then
      v_dtd_elements(1) := 'DRUGCHARACTERIZATION';
      v_dtd_elements(2) := 'MEDICINALPRODUCT';
      v_dtd_elements(3) := 'DRUGDOSAGEFORM';
    v_prod_name  := ESM_IMP.F_READ(:REPORT_ID,'MEDICINALPRODUCT');
      if v_prod_name is null then
         v_dtd_elements(2) := 'ACTIVESUBSTANCENAME';
      end if;
      lb_return := ESM_IMP.F_CHECK_AND_BUILD_KEY(:REPORT_ID,:PARENT_ELEMENT,v_dtd_elements);
      if (lb_return) then
          IF v_xml = 3 then
             l_number := 1;
          else
             l_number := -1;
          end if;

          if v_xml = 2 then
             l_number2 := 2;
          else
             l_number2 := 1;
          end if;
          l_return :=  ESM_IMP.F_WRITE(:REPORT_ID,:PARENT_ELEMENT,:DTD_ELEMENT,:PROFILE,'CASE_PROD_DRUGS','INTERACTION',l_number);
          l_return :=  ESM_IMP.F_WRITE(:REPORT_ID,:PARENT_ELEMENT,:DTD_ELEMENT,:PROFILE,'CASE_PRODUCT','DRUG_TYPE',l_number2);
      end if;
    end if;
END;
