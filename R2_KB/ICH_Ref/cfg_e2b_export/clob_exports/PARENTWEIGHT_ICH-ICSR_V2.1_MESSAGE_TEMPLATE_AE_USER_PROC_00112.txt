DECLARE
   v_xml varchar2(32767);
   l_return number := 0;
BEGIN
   v_xml := ESM_IMP.F_READ(:REPORT_ID,'PARENTWEIGHT');
   if v_xml is not null then
      l_return :=  ESM_IMP.F_WRITE(:REPORT_ID,:PARENT_ELEMENT,:DTD_ELEMENT,:PROFILE,'CASE_PARENT_INFO','WEIGHT_KG',v_xml);
      l_return :=  ESM_IMP.F_WRITE(:REPORT_ID,:PARENT_ELEMENT,:DTD_ELEMENT,:PROFILE,'CASE_PARENT_INFO','WEIGHT_LBS',v_xml*2.204999925);
   end if;
END;
