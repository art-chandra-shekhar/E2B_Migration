DECLARE
   v_xml1 varchar2(32767);
   v_xml2 varchar2(32767);
   v_xml3 varchar2(32767);
   v_xml4 varchar2(32767);
   l_return varchar2(4000) := null;
   l_return_code number := null;
   l_append NUMBER := 0;
   i NUMBER;
BEGIN
   v_xml1 := ESM_IMP.F_READ(:REPORT_ID,'PARENTLASTMENSTRUALDATE');
   v_xml2 := ESM_IMP.F_READ(:REPORT_ID,'PARENTLASTMENSTRUALDATEFORMAT');
   i := ESM_IMP_UTL.F_CHECK_UNIT_VALUE('PARENTLASTMENSTRUALDATE',v_xml1,'PARENTLASTMENSTRUALDATEFORMAT',v_xml2, :REPORT_ID);
   i := ESM_IMP_UTL.F_CHECK_VALIDATION(:PROFILE,'PARENTLASTMENSTRUALDATEFORMAT',v_xml2,:REPORT_ID);
   if v_xml1 is not null and v_xml2 is not null then  
      v_xml4 := ESM_AE.translate_dateformat(v_xml1,v_xml2,'N','PARENTLASTMENSTRUALDATE','PARENTLASTMENSTRUALDATEFORMAT',:REPORT_ID);
      v_xml3 := ESM_AE.translate_dateformat(v_xml1,v_xml2,'Y','PARENTLASTMENSTRUALDATE','PARENTLASTMENSTRUALDATEFORMAT',:REPORT_ID);
         if v_xml4 is not null then
            v_xml4 := to_char(to_date(v_xml4,'YYYYMMDDHH24MISS'),'DD-MON-YYYY HH24:MI:SS');
            l_return :=  ESM_IMP.F_WRITE(:REPORT_ID,:PARENT_ELEMENT,:DTD_ELEMENT,:PROFILE,'CASE_PARENT_INFO','DATE_OF_LMP',v_xml4);
            l_return :=  ESM_IMP.F_WRITE(:REPORT_ID,:PARENT_ELEMENT,:DTD_ELEMENT,:PROFILE,'CASE_PARENT_INFO','DATE_OF_LMP_PARTIAL',v_xml3);
         end if;
   end if;
   if v_xml2 is not null then
      l_return_code := ESM_IMP_UTL.F_CHECK_VALIDATION(:PROFILE,'PARENTLASTMENSTRUALDATEFORMAT',v_xml2,:REPORT_ID);
      if v_xml2 in ('102','203','204') then 
         v_xml2 := '8';
      elsif v_xml2 = 610 then 
         v_xml2 := '7';
      elsif v_xml2 = 602 then 
         v_xml2 := '5';
      else
         v_xml2 := NULL;
      end if;
      if v_xml2 is not NULL then
         l_return :=  ESM_IMP.F_WRITE(:REPORT_ID,:PARENT_ELEMENT,:DTD_ELEMENT,:PROFILE,'CASE_PARENT_INFO','DATE_OF_LMP_RES',v_xml2);
      end if;
   end if;
END;
