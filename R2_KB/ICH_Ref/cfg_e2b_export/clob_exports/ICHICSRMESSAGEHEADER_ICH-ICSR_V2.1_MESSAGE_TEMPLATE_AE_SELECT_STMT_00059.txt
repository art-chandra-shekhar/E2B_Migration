SELECT lower(cfg_message_type.root_element) root_element,profile_version,profile_release,case when :DB_UNIQUE_ID is not null then :MSG_ID || '_' || :DB_UNIQUE_ID  else to_char(:MSG_ID) end MESSAGENUMB, :MESSAGESENDERIDENTIFIER,:MESSAGERECEIVERIDENTIFIER,204,TO_NUMBER(TO_CHAR(ESM_UTL.GMT,'YYYYMMDDHH24MISS'),99999999999999) MESSAGEDATE FROM cfg_profile, cfg_message_type where cfg_profile.profile = :v_message_profile and cfg_message_type.id = :message_type_id
