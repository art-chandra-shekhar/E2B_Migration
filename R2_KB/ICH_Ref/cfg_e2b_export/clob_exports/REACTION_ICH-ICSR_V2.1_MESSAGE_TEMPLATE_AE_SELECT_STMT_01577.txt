SELECT SUBSTR(DESC_REPTD,1,200) DESC_REPTD,
  DECODE(INC_CODE,NULL,NULL,Esm_Utl.GET_MEDDRA_VERSION(CASE_EVENT.DICT_ID,3)) REACTIONMEDDRAVERSIONLLT,
  DECODE(Esm_Utl.GET_MEDDRA_VERSION(CASE_EVENT.DICT_ID,3),NULL,NULL,INC_CODE) REACTIONMEDDRALLT,
  DECODE(PREF_TERM,NULL,NULL,Esm_Utl.GET_MEDDRA_VERSION(CASE_EVENT.DICT_ID,3)) REACTIONMEDDRAVERSIONPT,
  DECODE(Esm_Utl.GET_MEDDRA_VERSION(CASE_EVENT.DICT_ID,3),NULL,NULL,PREF_TERM) REACTIONMEDDRAPT,
  Esm_Utl.F_TERMHIGHLIGHTED(:case_id,DECODE(RPT_SERIOUS,1,1,0,0,-1),case_event.seq_num) TERMHIGHLIGHTED,
  DECODE(ONSET,NULL,NULL,DECODE(ONSET_RES,5,602,7,610,8,DECODE(TO_CHAR(ONSET,'HH24:MI'),'00:00',102,203))) REACTIONSTARTDATEFORMAT,
  DECODE(ONSET_RES,5,TO_CHAR(ONSET,'YYYY'),7,TO_CHAR(ONSET,'YYYYMM'),8,DECODE(TO_CHAR(ONSET,'HH24:MI'),'00:00',TO_CHAR(ONSET,'YYYYMMDD'),TO_CHAR(ONSET,'YYYYMMDDHH24MI')),NULL) REACTIONSTARTDATE,
  DECODE(STOP_DATE,NULL,NULL,DECODE(STOP_DATE_RES,5,602,7,610,8,DECODE(TO_CHAR(STOP_DATE,'HH24:MI'),'00:00',102,203))) REACTIONENDDATEFORMAT,
  DECODE(STOP_DATE_RES,5,TO_CHAR(STOP_DATE,'YYYY') ,7,TO_CHAR(STOP_DATE,'YYYYMM'),8,DECODE(TO_CHAR(STOP_DATE,'HH24:MI'),'00:00',TO_CHAR(STOP_DATE,'YYYYMMDD'),TO_CHAR(STOP_DATE,'YYYYMMDDHH24MI')),NULL) REACTIONENDDATE,
  Esm_Utl.F_DURATION(duration_seconds,duration) REACTIONDURATION,
  Esm_Utl.F_DURATION_UNIT(duration_seconds,duration) REACTIONDURATIONUNIT,
  Esm_Utl.F_DURATION(onset_latency_seconds,onset_latency) REACTIONFIRSTIME,
  Esm_Utl.F_DURATION_UNIT(onset_latency_seconds,onset_latency) REACTIONFIRSTTIMEUNIT,
  Esm_Utl.F_DURATION(onset_delay_seconds,onset_delay) REACTIONLASTTIME,
  Esm_Utl.F_DURATION_UNIT(onset_delay_seconds,onset_delay) REACTIONLASTTIMEUNIT,
  DECODE(L.E2B_CODE,5,DECODE(CEA.UNRELATED,1,L.E2B_CODE,NULL),L.E2B_CODE) REACTIONOUTCOME
FROM CASE_EVENT,
  LM_EVT_OUTCOME L,
  (SELECT DISTINCT cea.case_id CEA_CASE_ID,
    cea.event_seq_num CEA_EVENT_SEQ_NUM,
    CASE
      WHEN ( DECODE (cea.rpt_causality_id,NULL, 0,-1, 0,lmc.reportability) = 1
      OR DECODE (cea.det_causality_id,NULL, 0,    -1, 0,lmc.reportability) = 1 )
      THEN 1
      ELSE 0
    END unrelated
  FROM case_event_assess cea,
    lm_causality lmc
  WHERE cea.case_id          = :CASE_ID
  AND cea.license_id         = 0
  AND ( cea.rpt_causality_id = lmc.causality_id
  OR cea.det_causality_id    = lmc.causality_id )
  AND (DECODE (cea.rpt_causality_id,NULL, 0,-1, 0,lmc.reportability) = 1
      OR DECODE (cea.det_causality_id,NULL, 0,    -1, 0,lmc.reportability) = 1)
  ) CEA
WHERE CASE_ID                 = :CASE_ID
AND CASE_EVENT.CASE_ID        = CEA.CEA_CASE_ID(+)
AND CASE_EVENT.SEQ_NUM        = CEA.CEA_EVENT_SEQ_NUM(+)
AND CASE_EVENT.EVT_OUTCOME_ID = L.EVT_OUTCOME_ID(+)
ORDER BY SORT_ID
